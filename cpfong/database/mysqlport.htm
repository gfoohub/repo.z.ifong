1. 系統需安裝libtool-1.3.3.tgz
    mount /cdrom
    pkg_add /cdrom/packages/All/libtool-1.3.3.tgz
  2. 切換目錄到mysql322-server
    cd /usr/ports/databases/mysql322-server
  3. 編輯Makefile
    ee Makefile
    MASTER_SITES=        ftp://ftp.center.kl.edu.tw/pub1/src/ \
    DB_DIR?= /usr/local/var
    USE_PERL5= yes
    USE_LIBTOOL= yes
    CONFIGURE_ARGS= --localstatedir=${DB_DIR} \
    --without-perl \
    --without-debug \
    --without-readline \
    --without-bench \
    --with-charset=big5 \
    --with-mysqld-user=mysql
  4. make NATIVE_THREADS=yes
  5. make install
  6. 執行初始化程式〈第一次安裝才需執行這個步驟〉
    cd     /usr/ports/databases/mysql322-server/work/mysql-3.22.32
    ./    scripts/mysql_install_db
  7. 接上頁 8.
